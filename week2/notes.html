2.3 Chess Board

						
const printChessBoard = (size: number): void => {
	if (size <= 0) {
		console.log('Invalid size.');
	}
	
	let board = '\n';
	for (let i = 0; i < size; i++) {
		let row = '';
		for (let j = 0; j < size; j++) {
			if (i % 2 !== j % 2) {
				row += '#';
			}

			row += ' ';
		}

		board += `${row}\n`;
	}

	console.log(board);
}

printChessBoard(8);

3.1 Minimum

						
	const min = (a: number, b: number): number => a > b ? b : a;

	console.log(min(0, 10));
	// → 0
	console.log(min(0, -10));
	// → -10

3.2 Recursion

						
	const isEven = (num: number): boolean => {
		if (num < 0) return isEven(num * -1);
		if (num === 0) return true;
		if (num === 1) return false;
		return isEven(num - 2);
	};

	console.log(isEven(50));
	// → true
	console.log(isEven(75));
	// → false
	console.log(isEven(-1));
	// → false

3.3 Bean Counting

						
const countChar = (haystack: string, needle: string, ignoreCase = true): number => {
	return haystack
		.split('')
		.map((char: string) => ignoreCase ? char.toLowerCase() : char)
		.filter((char: string) => char === needle)
		.length;
}

const countBs = (haystack: string): number => {
	return countChar(haystack, 'B', false);
}

// Note we're NOT ignoring the case and get 2 back
console.log(countBs("BBCbb"));
// → 2
console.log(countChar("kakkerlak", "k"));
// → 4
